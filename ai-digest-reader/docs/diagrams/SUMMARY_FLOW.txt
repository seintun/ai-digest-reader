================================================================================
                           SUMMARY STRUCTURE (v2)
================================================================================

    DigestSummary {
        simple: string           // "Major AI releases dominated today's feed..."
        themes: string[]        // ["Multimodal AI", "Extended context windows"]
        breaking: string[]      // ["Claude 4 released with 1M token context"]
        mustRead: MustReadItem[] // Curated essential articles
        fullBrief: string       // 2-3 paragraph detailed briefing
    }

    MustReadItem {
        title: string           // "Claude 4 Technical Report"
        url: string             // "https://anthropic.com/research/claude-4"
        source: 'reddit' | 'hn' // Source platform
        reason: string          // "Comprehensive deep-dive into the architecture"
    }

================================================================================
                           CLI OPTIONS
================================================================================

    python scripts/generator.py [options]

    Options:
      --no-ai              Skip AI summary generation
      --output-dir DIR     Output directory (default: output/)
      --limit N            Posts per source (default: 10)
      --subreddits ...     Filter subreddits
      --verbose            Verbose output
      --help               Show help
